let audioCtx; let scriptNode; let lastOut = 0; // 前回の出力値を覚えておく変数 const btn = document.getElementById('btn'); btn.addEventListener('click', () => { if (!audioCtx) { audioCtx = new (window.AudioContext || window.webkitAudioContext)(); } if (scriptNode) { scriptNode.disconnect(); scriptNode = null; btn.textContent = "Noise Start"; btn.style.background = "#333"; } else { scriptNode = audioCtx.createScriptProcessor(4096, 1, 1); scriptNode.onaudioprocess = (audioProcessingEvent) => { const outputBuffer = audioProcessingEvent.outputBuffer; const outputData = outputBuffer.getChannelData(0); for (let i = 0; i < outputBuffer.length; i++) { const white = Math.random() * 2 - 1; // 【ここがポイント！】 // 今回の音 = (前回の音 * 0.9) + (新しい乱数 * 0.1) // 90%くらい前の音を引き継ぐことで、急激な変化を抑えます。 const brown = (lastOut * 0.9) + (white * 0.1); outputData[i] = brown; lastOut = brown; // 今回の音を「次回用」に保存 } }; scriptNode.connect(audioCtx.destination); btn.textContent = "Noise Stop"; btn.style.background = "#e74c3c"; } });
